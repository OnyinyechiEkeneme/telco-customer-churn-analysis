//Period Table//

Time Period Table = 
VAR _max_date = MAX(Telco_Churn[SignupDate])
VAR _year_start = DATE(YEAR(_max_date), 1, 1)

RETURN
UNION(

    ADDCOLUMNS(
        CALENDAR(_year_start, _max_date),
        "Time Period", "YTD",
        "SortOrder", 1,
        "Period", "Year To Date"
    ),

    ADDCOLUMNS(
        CALENDAR(
            EOMONTH(_max_date, -2) + 1,
            EOMONTH(_max_date, -1)
        ),
        "Time Period", "PM",
        "SortOrder", 2,
        "Period", "Prior Month"
    ),

    ADDCOLUMNS(
        CALENDAR(
            EOMONTH(_max_date, -3) + 1,
            _max_date
        ),
        "Time Period", "3M",
        "SortOrder", 3,
        "Period", "3 Months"
    ),

    ADDCOLUMNS(
        CALENDAR(
            EOMONTH(_max_date, -6) + 1,
            _max_date
        ),
        "Time Period", "6M",
        "SortOrder", 4,
        "Period", "6 Months"
    ),

    ADDCOLUMNS(
        CALENDAR(
            EOMONTH(_max_date, -12) + 1,
            _max_date
        ),
        "Time Period", "12M",
        "SortOrder", 5,
        "Period", "12 Months"
    )
)
